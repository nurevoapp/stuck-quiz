const answers=new Array(20).fill(null);

let current=0;


const dimWeights=[

function startQuiz(){

  document.getElementById('intro-screen').style.display='none';

  const qs=document.getElementById('quiz-screen');

  qs.style.display='flex';

  renderQ();

}


function renderQ(){

  const q=Qs[current];

  document.getElementById('prog').style.width=((current/20)*100)+'%';

  document.getElementById('q-counter').textContent='Question '+(current+1)+' of 20';

  document.getElementById('q-text').textContent=q.q;

  const opts=document.getElementById('options');

  opts.innerHTML='';

  q.opts.forEach((o,i)=>{

    const b=document.createElement('button');

    b.className='opt'+(answers[current]===i?' selected':'');

    b.textContent=o;

    b.onclick=()=>select(i);

    opts.appendChild(b);

  });

  document.getElementById('btn-back').disabled=current===0;

  document.getElementById('btn-next').disabled=answers[current]===null;

  document.getElementById('btn-next').textContent=current===19?'See my results':'Next';

}


function select(i){

  answers[current]=i;

  document.querySelectorAll('.opt').forEach((b,idx)=>{

    b.classList.toggle('selected',idx===i);

  });

  document.getElementById('btn-next').disabled=false;

}


function goNext(){

  if(answers[current]===null)return;

  if(current===19){showResult();return;}

  current++;

  renderQ();

}


function goBack(){

  if(current===0)return;

  current--;

  renderQ();

}


function calcScores(){

  Object.keys(scores).forEach(k=>scores[k]=0);

  Qs.forEach((q,i)=>{

    if(answers[i]!==null){

      scores[q.dim]+=dimWeights[i][answers[i]];

    }

  });

}


function showResult(){

  calcScores();

  const maxDim=Object.entries(scores).sort((a,b)=>b[1]-a[1])[0][0];

  const t=TYPES[maxDim];

  const maxScore=9;

  document.getElementById('quiz-screen').style.display='none';

  const rs=document.getElementById('result-screen');

  rs.style.display='block';


  const barsHTML=Object.entries(scores).map(([k,v])=>{

    const pct=Math.round((v/maxScore)*100);

    const tc=TYPES[k];

    return `<div class="score-bar-wrap">

      <span class="score-label" style="font-size:11px">${tc.label.split(' ')[1]}</span>

      <div class="score-track"><div class="score-fill" style="width:${pct}%;background:${tc.color}"></div></div>

      <span class="score-val">${pct}%</span>

    </div>`;

  }).join('');


  rs.innerHTML=`

    <div style="text-align:center;margin-bottom:1.5rem">

      <div class="result-icon" style="background:${t.bg}">

        ${maxDim==='clarity'?'🔍':maxDim==='fear'?'😨':maxDim==='energy'?'⚡':maxDim==='strategy'?'🗺':maxDim==='environment'?'🌍':'🧠'}

      </div>

      <div class="result-type">${t.label}</div>

      <span class="result-tag" style="background:${t.bg};color:${t.color}">${maxDim.charAt(0).toUpperCase()+maxDim.slice(1)} block</span>

      <p class="result-desc">${t.desc}</p>

    </div>

    <div style="margin-bottom:1.25rem">

      <div style="font-size:12px;font-weight:500;color:#888;margin-bottom:10px">Your full profile</div>

    </div>

    <div class="email-gate">

      <p><strong style="font-size:13px;color:#1a1a18">Want your personalised action plan?</strong><br>Enter your email and we'll send you 5 specific steps to break through your ${maxDim} block — free.</p>

      <div class="email-row">

        <input class="email-input" type="email" id="email-in" placeholder="your@email.com">

      <div class="success-msg" id="success-msg">Done! Check your inbox — your action plan is on its way.</div>

    </div>

    <div style="text-align:center;margin-top:1rem">

      <button class="retake" onclick="retake()">Retake the quiz</button>

    </div>

  `;

  document.getElementById('prog').style.width='100%';

}


function submitEmail(dim){

  const val=document.getElementById('email-in').value.trim();

  if(!val||!val.includes('@'))return;

  document.getElementById('email-in').disabled=true;

  document.querySelector('.email-btn').disabled=true;

  document.getElementById('success-msg').style.display='block';

}


function retake(){

  Object.keys(scores).forEach(k=>scores[k]=0);

  answers.fill(null);

  current=0;

  document.getElementById('result-screen').style.display='none';

  document.getElementById('result-screen').innerHTML='';

  document.getElementById('quiz-screen').style.display='flex';

  renderQ();

}
